Chapter 3

  1. Use the str, vector, list, hash-map, and hash-set functions.
    2.           
            ;;with more than one arg,str returns the concatenation of the str values of the args.
            (str "Hello" "World")
            =>"HelloWorld"
            
            ;;A Vector is a collection of values indexed by contiguous integers.
            (vector 1 6 12)
            =>[1 6 12]

            ;;Lists are collections.
            (list 6 320 79)
            =>(6 320 79)

            ;;A Map is a collection that maps keys to values.
            (hash-map :a 1 :b 3 :c 8)
            => {c:8, b: 3, :a 1}

            ;;Sets are collections of unique values.
            (hash-set :j :t :c :f)
            => #{:j :t :c :f}
